16  package org.mybatis.caches.ehcache;
17  
18  import java.io.Serializable;
19  import java.nio.file.Path;
20  import java.time.Duration;
21  import java.util.concurrent.locks.ReadWriteLock;
22  
23  import org.apache.ibatis.cache.Cache;
24  import org.ehcache.PersistentCacheManager;
25  import org.ehcache.config.builders.CacheConfigurationBuilder;
26  import org.ehcache.config.builders.CacheManagerBuilder;
27  import org.ehcache.config.builders.ExpiryPolicyBuilder;
28  import org.ehcache.config.builders.ResourcePoolsBuilder;
29  import org.ehcache.config.units.EntryUnit;
30  import org.ehcache.config.units.MemoryUnit;
31  
32  /**
33   * Cache adapter for Ehcache 3.
34   *
35   * @author Simone Tripodi
36   */
37  public abstract class AbstractEhcacheCache implements Cache {
38  
39    /** Placeholder stored in Ehcache 3 for entries whose actual value is {@code null}. */
40    private static final Object NULL_VALUE = new NullValue();
41  
42    /**
43     * The cache manager reference. A {@link PersistentCacheManager} is used so that individual caches may optionally
44     * configure a disk tier via {@link #setMaxBytesLocalDisk(long)}.
45     */
46    protected static PersistentCacheManager CACHE_MANAGER = CacheManagerBuilder.newCacheManagerBuilder()
47        .with(
48            CacheManagerBuilder.persistence(Path.of(System.getProperty("java.io.tmpdir"), "ehcache-mybatis").toString()))
49        .build(true);
50  
51    /**
52     * The cache id (namespace).
53     */
54    protected final String id;
55  
56    /**
57     * The cache instance (lazily initialised on first use).
58     */
59    protected org.ehcache.Cache<Object, Object> cache;
60  
61    protected long timeToIdleSeconds;
62    protected long timeToLiveSeconds;
63    protected long maxEntriesLocalHeap;
64    protected long maxEntriesLocalDisk;
65    protected long maxBytesLocalDisk;
66    protected String memoryStoreEvictionPolicy;
67  
68    /**
69     * Instantiates a new abstract ehcache cache.
70     *
71     * @param id
72     *          the cache id (namespace)
73     */
74    public AbstractEhcacheCache(final String id) {
75      if (id == null) {
76        throw new IllegalArgumentException("Cache instances require an ID");
77      }
78      this.id = id;
79      // Remove any pre-existing cache so this instance always starts with a fresh default configuration.
80      if (CACHE_MANAGER.getCache(id, Object.class, Object.class) != null) {
81        CACHE_MANAGER.removeCache(id);
82      }
83    }
84  
85    /**
86     * Returns the underlying Ehcache 3 cache, creating it on first use with the current configuration.
87     */
88    protected synchronized org.ehcache.Cache<Object, Object> getOrCreateCache() {
89      if (cache == null) {
90        cache = buildAndRegisterCache();
91      }
92      return cache;
93    }
94  
95    /**
96     * Builds and registers a new Ehcache 3 cache instance using the current configuration fields.
97     */
98    protected org.ehcache.Cache<Object, Object> buildAndRegisterCache() {
99      if (CACHE_MANAGER.getCache(id, Object.class, Object.class) != null) {
100       CACHE_MANAGER.removeCache(id);
101     }
102     long heapEntries = maxEntriesLocalHeap > 0 ? maxEntriesLocalHeap : Long.MAX_VALUE / 2;
103     ResourcePoolsBuilder poolsBuilder = ResourcePoolsBuilder.newResourcePoolsBuilder().heap(heapEntries,
104         EntryUnit.ENTRIES);
105     if (maxBytesLocalDisk > 0) {
106       poolsBuilder = poolsBuilder.disk(maxBytesLocalDisk, MemoryUnit.B);
107     }
108     CacheConfigurationBuilder<Object, Object> builder = CacheConfigurationBuilder
109         .newCacheConfigurationBuilder(Object.class, Object.class, poolsBuilder).withExpiry(buildExpiryPolicy());
110     if (maxBytesLocalDisk > 0) {
111       // Disk and off-heap tiers require a Serializer since entries cannot be stored as object references.
112       // ObjectSerializer uses standard Java serialisation; cached values must implement Serializable.
113       builder = builder.withKeySerializer(ObjectSerializer.class).withValueSerializer(ObjectSerializer.class);
114     }
115     CACHE_MANAGER.createCache(id, builder.build());
116     return CACHE_MANAGER.getCache(id, Object.class, Object.class);
117   }
118 
119   private org.ehcache.expiry.ExpiryPolicy<Object, Object> buildExpiryPolicy() {
120     if (timeToLiveSeconds > 0) {
121       return ExpiryPolicyBuilder.timeToLiveExpiration(Duration.ofSeconds(timeToLiveSeconds));
122     }
123     if (timeToIdleSeconds > 0) {
124       return ExpiryPolicyBuilder.timeToIdleExpiration(Duration.ofSeconds(timeToIdleSeconds));
125     }
126     return ExpiryPolicyBuilder.noExpiration();
127   }
128 
129   /**
130    * {@inheritDoc}
131    */
132   @Override
133   public void clear() {
134     getOrCreateCache().clear();
135   }
136 
137   /**
138    * {@inheritDoc}
139    */
140   @Override
141   public String getId() {
142     return id;
143   }
144 
145   /**
146    * {@inheritDoc}
147    */
148   @Override
149   public Object getObject(Object key) {
150     Object value = getOrCreateCache().get(new HashKeyWrapper(key));
151     return value instanceof NullValue ? null : value;
152   }
153 
154   /**
155    * {@inheritDoc}
156    */
157   @Override
158   public int getSize() {
159     int count = 0;
160     for (org.ehcache.Cache.Entry<Object, Object> ignored : getOrCreateCache()) {
161       count++;
162     }
163     return count;
164   }
165 
166   /**
167    * {@inheritDoc}
168    */
169   @Override
170   public void putObject(Object key, Object value) {
171     getOrCreateCache().put(new HashKeyWrapper(key), value == null ? NULL_VALUE : value);
172   }
173 
174   /**
175    * {@inheritDoc}
176    */
177   @Override
178   public Object removeObject(Object key) {
179     Object obj = getObject(key);
180     getOrCreateCache().remove(new HashKeyWrapper(key));
181     return obj;
182   }
183 
184   /**
185    * {@inheritDoc}
186    */
187   public void unlock(Object key) {
188   }
189 
190   /**
191    * {@inheritDoc}
192    */
193   @Override
194   public boolean equals(Object obj) {
195     if (this == obj) {
196       return true;
197     }
198     if (obj == null) {
199       return false;
200     }
201     if (!(obj instanceof Cache)) {
202       return false;
203     }
204 
205     Cache otherCache = (Cache) obj;
206     return id.equals(otherCache.getId());
207   }
208 
209   /**
210    * {@inheritDoc}
211    */
212   @Override
213   public int hashCode() {
214     return id.hashCode();
215   }
216 
217   @Override
218   public ReadWriteLock getReadWriteLock() {
219     return null;
220   }
221 
222   /**
223    * {@inheritDoc}
224    */
225   @Override
226   public String toString() {
227     return "EHCache {" + id + "}";
228   }
229 
230   // DYNAMIC PROPERTIES
231 
232   /**
233    * Sets the time to idle for an element before it expires. Is only used if the element is not eternal. If the cache
234    * has already been initialised the configuration change takes effect immediately by recreating the cache.
235    *
236    * @param timeToIdleSeconds
237    *          the default amount of time to live for an element from its last accessed or modified date
238    */
239   public void setTimeToIdleSeconds(long timeToIdleSeconds) {
240     this.timeToIdleSeconds = timeToIdleSeconds;
241     recreateCacheIfInitialized();
242   }
243 
244   /**
245    * Sets the time to live for an element before it expires. Is only used if the element is not eternal. If the cache
246    * has already been initialised the configuration change takes effect immediately by recreating the cache.
247    *
248    * @param timeToLiveSeconds
249    *          the default amount of time to live for an element from its creation date
250    */
251   public void setTimeToLiveSeconds(long timeToLiveSeconds) {
252     this.timeToLiveSeconds = timeToLiveSeconds;
253     recreateCacheIfInitialized();
254   }
255 
256   /**
257    * Sets the maximum objects to be held in memory (0 = no limit). If the cache has already been initialised the
258    * configuration change takes effect immediately by recreating the cache.
259    *
260    * @param maxEntriesLocalHeap
261    *          The maximum number of elements in heap, before they are evicted (0 == no limit)
262    */
263   public void setMaxEntriesLocalHeap(long maxEntriesLocalHeap) {
264     this.maxEntriesLocalHeap = maxEntriesLocalHeap;
265     recreateCacheIfInitialized();
266   }
267 
268   /**
269    * Sets the maximum number elements on Disk. 0 means unlimited.
270    * <p>
271    * Note: this property is retained for compatibility but has no effect in Ehcache 3, which does not support an
272    * entry-count limit for the disk tier. Use {@link #setMaxBytesLocalDisk(long)} to configure disk storage instead.
273    * </p>
274    *
275    * @param maxEntriesLocalDisk
276    *          the maximum number of Elements to allow on the disk. 0 means unlimited.
277    */
278   public void setMaxEntriesLocalDisk(long maxEntriesLocalDisk) {
279     this.maxEntriesLocalDisk = maxEntriesLocalDisk;
280     recreateCacheIfInitialized();
281   }
282 
283   /**
284    * Sets the maximum bytes to be used for the disk tier. When greater than zero a disk resource pool is added to the
285    * cache, allowing entries evicted from the heap to overflow to disk. If set to zero (the default) no disk tier is
286    * configured and the cache is heap-only.
287    *
288    * @param maxBytesLocalDisk
289    *          the maximum number of bytes to allocate on disk. 0 means no disk tier (heap-only).
290    */
291   public void setMaxBytesLocalDisk(long maxBytesLocalDisk) {
292     this.maxBytesLocalDisk = maxBytesLocalDisk;
293     recreateCacheIfInitialized();
294   }
295 
296   /**
297    * Sets the eviction policy. Stored for informational purposes; Ehcache 3 manages its own eviction strategy.
298    *
299    * @param memoryStoreEvictionPolicy
300    *          a String representation of the policy. One of "LRU", "LFU" or "FIFO".
301    */
302   public void setMemoryStoreEvictionPolicy(String memoryStoreEvictionPolicy) {
303     this.memoryStoreEvictionPolicy = memoryStoreEvictionPolicy;
304     recreateCacheIfInitialized();
305   }
306 
307   /**
308    * Recreates the underlying Ehcache 3 cache with the current configuration if the cache has already been initialised.
309    * Called by property setters when a configuration change is requested after first use.
310    */
311   protected synchronized void recreateCacheIfInitialized() {
312     if (cache != null) {
313       cache = buildAndRegisterCache();
314     }
315   }
316 
317   /**
318    * Placeholder used to represent a cached {@code null} value. Ehcache 3 does not permit null values, so this sentinel
319    * is stored and translated back to {@code null} on retrieval.
320    */
321   private static final class NullValue implements Serializable {
322     private static final long serialVersionUID = 1L;
323   }
324
